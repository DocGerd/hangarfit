# 3D Layout & Tow-Fill Viewer Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add `hangarfit view` — a self-contained, offline, interactive 3D viewer of a hangar layout with a scrubbable whole-fill tow animation.

**Architecture:** A pure `scene.py` builder turns `Layout`/`MovesPlan`/`CheckResult` into a `hangarfit.scene/v1` JSON dict; `viewer.py` inlines that scene + vendored Three.js into one double-clickable HTML; `_viewer_assets/viewer.js` is a thin Three.js consumer. All transform math (the ADR-0002 det-−1 trap) stays in Python — the viewer only applies per-frame 2×3 affine matrices.

**Tech Stack:** Python 3.12, shapely (existing), Three.js (vendored, ESM, `data:`-import-map), no JS build step.

**Spec:** `docs/superpowers/specs/2026-06-03-3d-viewer-design.md`. Architecture checkpoint #1 confirmed the affine reproduces `aircraft_parts_world` to 1.8e-15.

```python
# src/hangarfit/scene.py
"""Pure builder for the ``hangarfit.scene/v1`` JSON contract.

Turns a :class:`~hangarfit.models.Layout` (+ optional ``MovesPlan`` and
``CheckResult``) into a JSON-serializable dict consumed by the offline
Three.js viewer (:mod:`hangarfit.viewer`). No I/O, no rendering: every
geometry/transform value is computed here so the viewer never re-derives
the determinant-−1 plane-local→world transform (ADR-0002, ADR-0017).
"""
from __future__ import annotations

import math
from typing import TYPE_CHECKING

from hangarfit.geometry import aircraft_parts_world
from hangarfit.models import CheckResult, Layout
from hangarfit.towplanner import back_first_order
from hangarfit.visualize import PLANES

if TYPE_CHECKING:
    from hangarfit.towplanner import MovesPlan

SCHEMA = "hangarfit.scene/v1"
Affine = tuple[float, float, float, float, float, float]


def _color_map(plane_ids: list[str]) -> dict[str, str]:
    """Stable per-plane colour by sorted id — parity with visualize._plane_colour_map."""
    ordered = sorted(set(plane_ids))
    return {pid: PLANES[i % len(PLANES)] for i, pid in enumerate(ordered)}


def _hangar_block(layout: Layout) -> dict:
    h = layout.hangar
    bay = h.maintenance_bay
    return {
        "length_m": h.length_m,
        "width_m": h.width_m,
        "door": {"center_x_m": h.door.center_x_m, "width_m": h.door.width_m},
        "maintenance_bay": {
            "center_x_m": bay.center_x_m,
            "width_m": bay.width_m,
            "depth_m": bay.depth_m,
            "closed": layout.maintenance_plane is not None,
            "plane_id": layout.maintenance_plane,
        },
    }
```

```python
def _plane_blocks(layout: Layout) -> list[dict]:
    """One block per placed plane: id, colour, and plane-local box list.

    A box mirrors a Part: plane-local centre (cx=forward offset, cy=right
    offset, cz=mid-height), extents (length along +x, width along +y, height
    along z), and ``angle_deg`` (CCW within plane-local, as oriented_rect uses).
    """
    colour = _color_map([p.plane_id for p in layout.placements])
    blocks: list[dict] = []
    for placement in sorted(layout.placements, key=lambda p: p.plane_id):
        ac = layout.fleet[placement.plane_id]
        boxes = [
            {
                "kind": part.kind,
                "cx": part.offset_x_m,
                "cy": part.offset_y_m,
                "cz": (part.z_top_m + part.z_bottom_m) / 2.0,
                "length_m": part.length_m,
                "width_m": part.width_m,
                "height_m": part.z_top_m - part.z_bottom_m,
                "angle_deg": part.angle_deg,
            }
            for part in ac.parts
        ]
        blocks.append({"id": placement.plane_id, "color": colour[placement.plane_id], "boxes": boxes})
    return blocks
```

```python
from hangarfit.geometry import oriented_rect  # add to imports
from hangarfit.models import Placement          # add to imports


def _affine(placement: Placement) -> Affine:
    """The plane-local→world affine (a,b,tx,c,d,ty), matching geometry.local_to_world.

    For a plane-local point (u=forward, v=right):
        world_x = u·sin h + v·cos h + x        world_y = u·cos h − v·sin h + y
    The linear part has determinant −1 (ADR-0002). The viewer applies this as a
    THREE.Matrix4 to a statically-built plane-local group — it does no transform
    math of its own.
    """
    h = math.radians(placement.heading_deg)
    s, c = math.sin(h), math.cos(h)
    return (s, c, placement.x_m, c, -s, placement.y_m)


def _anchors(layout: Layout) -> dict[str, list[list[list[float]]]]:
    """Per-plane, per-box world corner points at the FINAL placement, from the
    production oracle. The viewer recomputes these from its box geometry + the
    final affine and asserts agreement on load (fail-loud transform check)."""
    out: dict[str, list[list[list[float]]]] = {}
    for placement in sorted(layout.placements, key=lambda p: p.plane_id):
        ac = layout.fleet[placement.plane_id]
        out[placement.plane_id] = [
            [[x, y] for x, y in list(wp.polygon.exterior.coords)[:-1]]
            for wp in aircraft_parts_world(ac, placement)
        ]
    return out
```

```python
from hangarfit.towplanner import back_first_order  # add to imports


def _pose_affine(x_m: float, y_m: float, heading_deg: float) -> Affine:
    h = math.radians(heading_deg)
    s, c = math.sin(h), math.cos(h)
    return (s, c, x_m, c, -s, y_m)


def _sample_affines(path, max_samples: int) -> list[Affine]:
    # Coarsen step so a long path never blows past max_samples (keeps HTML small).
    length = path.length_m
    step_m = max(0.05, length / max_samples) if length > 0 else 0.05
    step_deg = max(1.0, 360.0 / max_samples)
    poses = list(path.sample(step_m=step_m, step_deg=step_deg))
    return [_pose_affine(p.x_m, p.y_m, p.heading_deg) for p in poses]


def _timeline(
    layout: Layout,
    moves_plan: "MovesPlan | None",
    *,
    tow_speed_mps: float = 1.0,
    min_seg_s: float = 1.5,
    max_seg_s: float = 6.0,
    max_samples_per_path: int = 240,
) -> tuple[dict, dict[str, Affine]]:
    """Build the sequential whole-fill timeline + per-plane final affines.

    Static (no plan): segments == [], total_s == 0, finals = every plane parked.
    """
    finals: dict[str, Affine] = {
        p.plane_id: _pose_affine(p.x_m, p.y_m, p.heading_deg) for p in layout.placements
    }
    if moves_plan is None:
        return {"total_s": 0.0, "segments": []}, finals

    move_by_id = {m.plane_id: m for m in moves_plan.moves}
    segments: list[dict] = []
    t = 0.0
    for placement in back_first_order(layout.placements):
        move = move_by_id.get(placement.plane_id)
        if move is None:
            continue  # defensive: a placement with no move stays at its final pose
        samples = _sample_affines(move.path, max_samples_per_path)
        dur = min(max(move.path.length_m / tow_speed_mps, min_seg_s), max_seg_s)
        segments.append(
            {"plane_id": placement.plane_id, "start_s": t, "end_s": t + dur, "samples": samples}
        )
        finals[placement.plane_id] = samples[-1]
        t += dur
    return {"total_s": t, "segments": segments}, finals
```

```python
_COORD_NOTE = (
    "world: x right along door wall, y deeper into hangar, z up; "
    "heading_deg compass-style (from +y, CW+). See ADR-0002 / ADR-0017."
)


def _conflict_ids(check_result: CheckResult | None) -> list[str]:
    if check_result is None:
        return []
    return sorted({pid for c in check_result.conflicts for pid in c.planes})


def build_scene(
    layout: Layout,
    *,
    moves_plan: "MovesPlan | None" = None,
    check_result: CheckResult | None = None,
    tow_speed_mps: float = 1.0,
    min_seg_s: float = 1.5,
    max_seg_s: float = 6.0,
    max_samples_per_path: int = 240,
) -> dict:

    timeline, finals = _timeline(
        layout, moves_plan, tow_speed_mps=tow_speed_mps,
        min_seg_s=min_seg_s, max_seg_s=max_seg_s, max_samples_per_path=max_samples_per_path,

        "schema": SCHEMA,
        "units": "m",
        "coordinate_note": _COORD_NOTE,
        "hangar": _hangar_block(layout),
        "planes": _plane_blocks(layout),
        "timeline": timeline,

        "conflicts": _conflict_ids(check_result),
        "anchors": _anchors(layout),
```

```python
# src/hangarfit/viewer.py
"""Assemble a self-contained, offline 3D viewer HTML from a scene/v1 dict.

Inlines the scene JSON + a ``data:`` URL import-map for the vendored Three.js
so a double-clicked ``file://`` page resolves ``import ... from 'three'`` with
zero network (sidesteps the ES-module file:// CORS block without a server).
See ADR-0017.
"""
from __future__ import annotations

import base64
import json
from importlib import resources
from pathlib import Path

_ASSETS = "hangarfit._viewer_assets"
_THREE = "hangarfit._viewer_assets.three"


def _asset_text(pkg: str, name: str) -> str:
    return resources.files(pkg).joinpath(name).read_text(encoding="utf-8")


def _data_url(js_source: str) -> str:
    b64 = base64.b64encode(js_source.encode("utf-8")).decode("ascii")
    return f"data:text/javascript;base64,{b64}"


def render_viewer(scene: dict, output_path: Path | str) -> None:
    """Write a single self-contained offline HTML viewer for ``scene``."""
    three_src = _asset_text(_THREE, "three.module.js")
    orbit_src = _asset_text(_THREE, "OrbitControls.js")
    viewer_js = _asset_text(_ASSETS, "viewer.js")
```

> **Local-frame note (critical):** The group matrix is the 2D affine on
> `(u, v)` = plane-local `(forward, right)`. In Three.js a `Matrix4` acts on
> `(X, Y, Z)`. We bind **local X = u (forward), local Y = v (right), local Z = w
> (height)** so the affine's `(a,b,tx,c,d,ty)` map columns line up: row0 →
> world_x from (X=u, Y=v), row1 → world_y from (X=u, Y=v), row2 → world_z = Z.
> THREE's world "up" must therefore be **+Z**, not the default +Y. Two options —
> pick ONE and keep it consistent:
> (A) Keep THREE's default +Y up but **emit the affine into Y/… columns** — messy.
> (B) **Use +Z as world up** for the whole scene (camera, lights, hangar built in
> x-right / y-deep / z-up), so box height is along world Z and the affine maps
> `(u,v)`→world `(x,y)` with z untouched. **Choose (B).**
>
> Under (B): build boxes with `BoxGeometry(length_m /*X=u*/, width_m /*Y=v*/,
> height_m /*Z=w*/)`, `mesh.position.set(cx, cy, cz)`, rotate `angle_deg` (CCW)
> about **local Z**: `mesh.rotation.z = THREE.MathUtils.degToRad(b.angle_deg)`.
> Set `scene.up`/`cam.up = (0,0,1)`, build the hangar floor in the XY plane at
> z=0 (no `rotation.x`), walls extruded in +Z, grid rotated into XY. The affine
> matrix then is exactly `affineMatrix([a,b,tx,c,d,ty])` as written (row2 = Z
> identity). **Rewrite `addHangar()` and the box build for +Z-up before wiring
> the anchor check** — the anchor self-check will catch any mismatch.

```javascript
function boxCornersLocal(b) {
  const h = THREE.MathUtils.degToRad(b.angle_deg), cs = Math.cos(h), sn = Math.sin(h);
  const hl = b.length_m / 2, hw = b.width_m / 2;
  // oriented_rect corner order: (+hl,-hw),(+hl,+hw),(-hl,+hw),(-hl,-hw) about (cx,cy)
  return [[hl, -hw], [hl, hw], [-hl, hw], [-hl, -hw]].map(([lx, ly]) =>
    [b.cx + lx * cs - ly * sn, b.cy + lx * sn + ly * cs]); // CCW rot in (u,v)
}
function applyAffine([a, b, tx, c, d, ty], u, v) { return [a * u + b * v + tx, c * u + d * v + ty]; }
(function checkAnchors() {
  let maxErr = 0;
  for (const p of SCENE.planes) {
    const aff = SCENE.final_poses[p.id], want = SCENE.anchors[p.id];
    p.boxes.forEach((b, bi) => boxCornersLocal(b).forEach(([u, v], ci) => {
      const [wx, wy] = applyAffine(aff, u, v);
      maxErr = Math.max(maxErr, Math.abs(wx - want[bi][ci][0]), Math.abs(wy - want[bi][ci][1]));
    }));

    '): viewer affine disagrees with the Python oracle. Do not trust this render.');
})();

  if (!seg) return { vis: true, aff: SCENE.final_poses[pid] };       // static scene
  if (t < seg.start_s) return { vis: false, aff: null };             // not entered yet
  if (t >= seg.end_s) return { vis: true, aff: SCENE.final_poses[pid] };
  const frac = (t - seg.start_s) / (seg.end_s - seg.start_s);
  const i = Math.round(frac * (seg.samples.length - 1));
  return { vis: true, aff: seg.samples[i] };
}
```
